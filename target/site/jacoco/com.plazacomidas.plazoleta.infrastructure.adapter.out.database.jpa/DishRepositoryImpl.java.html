package com.plazacomidas.plazoleta.infrastructure.adapter.out.database.jpa;

import com.plazacomidas.plazoleta.domain.model.Dish;
import com.plazacomidas.plazoleta.domain.repository.DishRepository;
import com.plazacomidas.plazoleta.infrastructure.adapter.out.database.jpa.entity.DishEntity;
import com.plazacomidas.plazoleta.infrastructure.adapter.out.database.jpa.mapper.DishEntityMapper;
import com.plazacomidas.plazoleta.infrastructure.adapter.out.database.jpa.repository.DishJpaRepository;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Repository;

import java.util.Optional;

@Repository
@RequiredArgsConstructor
public class DishRepositoryImpl implements DishRepository {

    private final DishJpaRepository jpaRepository;
    private final DishEntityMapper mapper;

    @Override
    public Dish save(Dish dish) {
        DishEntity entity = mapper.toEntity(dish);
        DishEntity saved = jpaRepository.save(entity);
        return mapper.toDomain(saved);
    }

    @Override
    public void update(Dish dish) {
        DishEntity entityToUpdate = mapper.toEntity(dish);
        jpaRepository.save(entityToUpdate);
    }

    @Override
    public Optional<Dish> findById(Long id) {
        return jpaRepository.findById(id).map(mapper::toDomain);
    }
}
